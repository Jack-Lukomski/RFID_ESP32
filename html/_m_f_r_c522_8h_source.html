181// MFRC522's commands for the PCD.
182#define PCD_CMD_IDLE 0x00 // NO action; cancels current command execution
183#define PCD_CMD_MEM 0x01 // Stores 25 bytes into the internal buffer
184#define PCD_CMD_GEN_RANDOM_ID 0x02 // Generates a 10-byte random ID number
185#define PCD_CMD_CALC_CRC 0x03 // Activates the CRC coprocessor or performs a self-test
186#define PCD_CMD_TRANSMIT 0x04 // Transmits data from the FIFO buffer
187#define PCD_CMD_NO_CMD_CHANGE 0x07 // Can be used to modify the CommandReg register bits without affecting the command, if any, currently being executed
188#define PCD_CMD_RECEIVE 0x08 // Activates the receiver circuits
189#define PCD_CMD_TRANSCEIVE 0x0C // Transmits data from FIFO buffer to antenna and automatically activates the receiver after transmission
190#define PCD_CMD_MF_AUTHENT 0x0E // Performs the MIFARE standard authentication as a reader
191#define PCD_CMD_SOFT_RESET 0x0F // Resets the MFRC522
