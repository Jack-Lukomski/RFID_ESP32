112#define PCD_CMD_MEM 0x01 // Stores 25 bytes into the internal buffer
113#define PCD_CMD_GEN_RANDOM_ID 0x02 // Generates a 10-byte random ID number
114#define PCD_CMD_CALC_CRC 0x03 // Activates the CRC coprocessor or performs a self-test
115#define PCD_CMD_TRANSMIT 0x04 // Transmits data from the FIFO buffer
116#define PCD_CMD_NO_CMD_CHANGE 0x07 // Can be used to modify the CommandReg register bits without affecting the command, if any, currently being executed
117#define PCD_CMD_RECEIVE 0x08 // Activates the receiver circuits
118#define PCD_CMD_TRANSCEIVE 0x0C // Transmits data from FIFO buffer to antenna and automatically activates the receiver after transmission
119#define PCD_CMD_MF_AUTHENT 0x0E // Performs the MIFARE standard authentication as a reader
